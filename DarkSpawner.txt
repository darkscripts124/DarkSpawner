local function detectExecutor()
    local name = "Unknown"

    if identifyexecutor then
        local success, exec = pcall(identifyexecutor)
        if success and exec then
            name = exec
        end
    elseif getexecutorname then
        local success, exec = pcall(getexecutorname)
        if success and exec then
            name = exec
        end
    elseif getgenv().isdelta or getgenv().delta then
        name = "Delta"
    elseif is_sirhurt_closure then
        name = "SirHurt"
    elseif pebc_execute then
        name = "ProtoSmasher"
    elseif KRNL_LOADED then
        name = "KRNL"
    elseif syn and syn.protect_gui then
        name = "Synapse X"
    elseif fluxus then
        name = "Fluxus"
    elseif isexecutorclosure then
        name = "Script-Ware"
    elseif getgenv().is_comet then
        name = "Comet"
    elseif getgenv().Valyse then
        name = "Valyse"
    elseif getgenv().dx9ware then
        name = "DX9Ware"
    end

    return name
end

-- Detect executor
local execName = detectExecutor()

-- Kick if using Delta
if execName == "Delta" then
    game:GetService("Players").LocalPlayer:Kick("Your executor is not supported with this script.")
else
    -- Optional: Notify the user of allowed executor
    pcall(function()
        game.StarterGui:SetCore("SendNotification", {
            Title = "Executor Detected";
            Text = "You're using: " .. execName;
            Duration = 5;
        })
    end)
end
